#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);  // Number of test cases

    for (int t = 0; t < n; t++) {
        int num_towers, height;
        scanf("%d %d", &num_towers, &height);

        if (height == 1 || num_towers % 2 == 0)
            printf("2\n");
        else
            printf("1\n");
    }

    return 0;
}
