#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int count[101] = {0}; // Array to store frequency of numbers

    // Read numbers and count occurrences
    for (int i = 0; i < n; i++) {
        int num;
        scanf("%d", &num);
        count[num]++;
    }

    int totalPairs = 0;

    // Calculate pairs
    for (int i = 0; i <= 100; i++) {
        totalPairs += count[i] / 2; // Each pair needs 2 items
    }

    printf("%d\n", totalPairs);

    return 0;
}
