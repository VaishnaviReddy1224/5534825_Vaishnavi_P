#include <stdio.h>

int main() {
    int size;
    scanf("%d", &size); 

    int counts[100] = {0}; 

    for (int i = 0; i < size; i++) {
        int number;
        scanf("%d", &number); 
        counts[number]++;      
    }

   
    for (int i = 0; i < 100; i++) {
        printf("%d ", counts[i]);
    }

    return 0;
}
