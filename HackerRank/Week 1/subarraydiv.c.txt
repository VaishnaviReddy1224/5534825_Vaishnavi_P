#include <stdio.h>
#include <stdlib.h>

int main() {
    int n;
    scanf("%d", &n);

    int squares[n];
    for (int i = 0; i < n; i++)
        scanf("%d", &squares[i]);

    int d, m;
    scanf("%d %d", &d, &m);

    int count = 0;

    // Check all subarrays of length m
    for (int i = 0; i <= n - m; i++) {
        int sum = 0;
        for (int j = 0; j < m; j++)
            sum += squares[i + j];
        if (sum == d)
            count++;
    }

    printf("%d\n", count);

    return 0;
}
