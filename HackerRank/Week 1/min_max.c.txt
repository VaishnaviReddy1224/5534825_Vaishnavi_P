#include <stdio.h>

int main() {
    int numbers[5];
    long total = 0;
    int smallest, largest;

    
    for (int i = 0; i < 5; i++) {
        scanf("%d", &numbers[i]);
        total += numbers[i];
    }

    smallest = largest = numbers[0];

   
    for (int i = 1; i < 5; i++) {
        if (numbers[i] < smallest)
            smallest = numbers[i];
        if (numbers[i] > largest)
            largest = numbers[i];
    }

    
    printf("%ld %ld", total - largest, total - smallest);
    
    return 0;
}
