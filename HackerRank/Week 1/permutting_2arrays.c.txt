#include <stdio.h>
#include <stdlib.h>

int compare(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int n, k;
        scanf("%d %d", &n, &k);

        int a[n], b[n];
        for (int i = 0; i < n; i++)
            scanf("%d", &a[i]);
        for (int i = 0; i < n; i++)
            scanf("%d", &b[i]);

        // Sort a in ascending, b in ascending
        qsort(a, n, sizeof(int), compare);
        qsort(b, n, sizeof(int), compare);

        int possible = 1;
        for (int i = 0; i < n; i++) {
            if (a[i] + b[n - i - 1] < k) {
                possible = 0;
                break;
            }
        }

        if (possible)
            printf("YES\n");
        else
            printf("NO\n");
    }

    return 0;
}
