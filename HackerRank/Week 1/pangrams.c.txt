#include <stdio.h>

int main() {
    char str[1001];
    int mark[26] = {0};  // To track letters a-z
    int i;

    fgets(str, sizeof(str), stdin);  // Read the input string

    // Check each character
    for (i = 0; str[i] != '\0'; i++) {
        char c = str[i];
        if (c >= 'a' && c <= 'z')
            mark[c - 'a'] = 1;
        else if (c >= 'A' && c <= 'Z')
            mark[c - 'A'] = 1;
    }

    // Check if all letters are present
    for (i = 0; i < 26; i++) {
        if (mark[i] == 0)
            break;
    }

    if (i == 26)
        printf("pangram\n");
    else
        printf("not pangram\n");

    return 0;
}
