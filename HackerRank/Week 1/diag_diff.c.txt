#include <stdio.h>
#include <stdlib.h> 

int main() {
    int n;
    scanf("%d", &n); 

    int matrix[100][100];
    int diagonal1 = 0, diagonal2 = 0;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &matrix[i][j]); 

            if (i == j)
                diagonal1 += matrix[i][j];           
            if (i + j == n - 1)
                diagonal2 += matrix[i][j];           
        }
    }

    int difference = abs(diagonal1 - diagonal2);   
    printf("%d\n", difference);

    return 0;
}
