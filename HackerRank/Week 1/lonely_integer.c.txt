#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);   // read number of elements

    int arr[100];      // assume max 100 numbers
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    // Find the number that appears only once
    for (int i = 0; i < n; i++) {
        int count = 0;
        for (int j = 0; j < n; j++) {
            if (arr[i] == arr[j]) {
                count++;
            }
        }
        if (count == 1) {
            printf("%d\n", arr[i]);
            break;
        }
    }

    return 0;
}
