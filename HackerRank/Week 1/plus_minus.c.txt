#include <stdio.h>

void plusMinus(int arr[], int size) {
    int pos = 0, neg = 0, zero = 0;

    for(int i = 0; i < size; i++) {
        if(arr[i] > 0)
            pos++;
        else if(arr[i] < 0)
            neg++;
        else
            zero++;
    }

    printf("%.6f\n", (float)pos / size);
    printf("%.6f\n", (float)neg / size);
    printf("%.6f\n", (float)zero / size);
}

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    plusMinus(arr, n);

    return 0;
}
